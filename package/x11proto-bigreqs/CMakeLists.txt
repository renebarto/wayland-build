project(bigreqsproto)

set(X11PROTO_BIGREQS_VERSION 1.1.2)
set(X11PROTO_BIGREQS_REQUIRED_VERSION ${X11PROTO_BIGREQS_VERSION})

find_package(X11ProtoBigReqs QUIET)

set(X11PROTO_BIGREQS_NAME ${PROJECT_NAME})
set(X11PROTO_BIGREQS_BASE ${X11PROTO_BIGREQS_NAME}-${X11PROTO_BIGREQS_VERSION})

if (NOT X11PROTO_BIGREQS_FOUND)
    set(X11PROTO_BIGREQS_EXTENSION tar.gz)
    set(X11PROTO_BIGREQS_FULLNAME ${X11PROTO_BIGREQS_BASE}.${X11PROTO_BIGREQS_EXTENSION})
    set(X11PROTO_BIGREQS_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_BIGREQS_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_BIGREQS_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_BIGREQS_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_BIGREQS_SITE}
        ${X11PROTO_BIGREQS_FULLNAME}
        ${X11PROTO_BIGREQS_BASE}
        ${X11PROTO_BIGREQS_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}")
else()
    add_custom_target(${X11PROTO_BIGREQS_BASE})
endif()
