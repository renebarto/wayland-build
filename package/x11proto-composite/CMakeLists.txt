project(compositeproto)

set(X11PROTO_COMPOSITE_VERSION 0.4.2)
set(X11PROTO_COMPOSITE_REQUIRED_VERSION ${X11PROTO_COMPOSITE_VERSION})

find_package(X11ProtoComposite QUIET)

set(X11PROTO_COMPOSITE_NAME ${PROJECT_NAME})
set(X11PROTO_COMPOSITE_BASE ${X11PROTO_COMPOSITE_NAME}-${X11PROTO_COMPOSITE_VERSION})

if (NOT X11PROTO_COMPOSITE_FOUND)
    set(X11PROTO_COMPOSITE_EXTENSION tar.gz)
    set(X11PROTO_COMPOSITE_FULLNAME ${X11PROTO_COMPOSITE_BASE}.${X11PROTO_COMPOSITE_EXTENSION})
    set(X11PROTO_COMPOSITE_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_COMPOSITE_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_COMPOSITE_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_COMPOSITE_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_COMPOSITE_SITE}
        ${X11PROTO_COMPOSITE_FULLNAME}
        ${X11PROTO_COMPOSITE_BASE}
        ${X11PROTO_COMPOSITE_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}")
else()
    add_custom_target(${X11PROTO_COMPOSITE_BASE})
endif()
