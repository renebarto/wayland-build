project(damageproto)

set(X11PROTO_DAMAGE_VERSION 1.2.1)
set(X11PROTO_DAMAGE_REQUIRED_VERSION ${X11PROTO_DAMAGE_VERSION})

find_package(X11ProtoDamage QUIET)

set(X11PROTO_DAMAGE_NAME ${PROJECT_NAME})
set(X11PROTO_DAMAGE_BASE ${X11PROTO_DAMAGE_NAME}-${X11PROTO_DAMAGE_VERSION})

if (NOT X11PROTO_DAMAGE_FOUND)
    set(X11PROTO_DAMAGE_EXTENSION tar.gz)
    set(X11PROTO_DAMAGE_FULLNAME ${X11PROTO_DAMAGE_BASE}.${X11PROTO_DAMAGE_EXTENSION})
    set(X11PROTO_DAMAGE_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_DAMAGE_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_DAMAGE_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_DAMAGE_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_DAMAGE_SITE}
        ${X11PROTO_DAMAGE_FULLNAME}
        ${X11PROTO_DAMAGE_BASE}
        ${X11PROTO_DAMAGE_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}"
        "make"
        "make check"
        "make install")
else()
    add_custom_target(${X11PROTO_DAMAGE_BASE})
endif()
