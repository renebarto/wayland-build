project(inputproto)

set(X11PROTO_INPUT_VERSION 2.3.2)
set(X11PROTO_INPUT_REQUIRED_VERSION ${X11PROTO_INPUT_VERSION})

find_package(X11ProtoInput QUIET)

set(X11PROTO_INPUT_NAME ${PROJECT_NAME})
set(X11PROTO_INPUT_BASE ${X11PROTO_INPUT_NAME}-${X11PROTO_INPUT_VERSION})

if (NOT X11PROTO_INPUT_FOUND)
    set(X11PROTO_INPUT_EXTENSION tar.gz)
    set(X11PROTO_INPUT_FULLNAME ${X11PROTO_INPUT_BASE}.${X11PROTO_INPUT_EXTENSION})
    set(X11PROTO_INPUT_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_INPUT_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_INPUT_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_INPUT_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_INPUT_SITE}
        ${X11PROTO_INPUT_FULLNAME}
        ${X11PROTO_INPUT_BASE}
        ${X11PROTO_INPUT_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}")
else()
    add_custom_target(${X11PROTO_INPUT_BASE})
endif()
