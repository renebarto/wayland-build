project(recordproto)

set(X11PROTO_RECORD_VERSION 1.14.2)
set(X11PROTO_RECORD_REQUIRED_VERSION ${X11PROTO_RECORD_VERSION})

find_package(X11ProtoRecord QUIET)

set(X11PROTO_RECORD_NAME ${PROJECT_NAME})
set(X11PROTO_RECORD_BASE ${X11PROTO_RECORD_NAME}-${X11PROTO_RECORD_VERSION})

if (NOT X11PROTO_RECORD_FOUND)
    set(X11PROTO_RECORD_EXTENSION tar.gz)
    set(X11PROTO_RECORD_FULLNAME ${X11PROTO_RECORD_BASE}.${X11PROTO_RECORD_EXTENSION})
    set(X11PROTO_RECORD_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_RECORD_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_RECORD_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_RECORD_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_RECORD_SITE}
        ${X11PROTO_RECORD_FULLNAME}
        ${X11PROTO_RECORD_BASE}
        ${X11PROTO_RECORD_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}"
        "make"
        "make check"
        "make install")
else()
    add_custom_target(${X11PROTO_RECORD_BASE})
endif()
