project(renderproto)

set(X11PROTO_RENDER_VERSION 0.11.1)
set(X11PROTO_RENDER_REQUIRED_VERSION ${X11PROTO_RENDER_VERSION})

find_package(X11ProtoRender QUIET)

set(X11PROTO_RENDER_NAME ${PROJECT_NAME})
set(X11PROTO_RENDER_BASE ${X11PROTO_RENDER_NAME}-${X11PROTO_RENDER_VERSION})

if (NOT X11PROTO_RENDER_FOUND)
    set(X11PROTO_RENDER_EXTENSION tar.gz)
    set(X11PROTO_RENDER_FULLNAME ${X11PROTO_RENDER_BASE}.${X11PROTO_RENDER_EXTENSION})
    set(X11PROTO_RENDER_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_RENDER_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_RENDER_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_RENDER_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_RENDER_SITE}
        ${X11PROTO_RENDER_FULLNAME}
        ${X11PROTO_RENDER_BASE}
        ${X11PROTO_RENDER_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}"
        "make"
        "make check"
        "make install")
else()
    add_custom_target(${X11PROTO_RENDER_BASE})
endif()
