project(resourceproto)

set(X11PROTO_RESOURCE_VERSION 1.2.0)
set(X11PROTO_RESOURCE_REQUIRED_VERSION ${X11PROTO_RESOURCE_VERSION})

find_package(X11ProtoResource QUIET)

set(X11PROTO_RESOURCE_NAME ${PROJECT_NAME})
set(X11PROTO_RESOURCE_BASE ${X11PROTO_RESOURCE_NAME}-${X11PROTO_RESOURCE_VERSION})

if (NOT X11PROTO_RESOURCE_FOUND)
    set(X11PROTO_RESOURCE_EXTENSION tar.gz)
    set(X11PROTO_RESOURCE_FULLNAME ${X11PROTO_RESOURCE_BASE}.${X11PROTO_RESOURCE_EXTENSION})
    set(X11PROTO_RESOURCE_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_RESOURCE_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_RESOURCE_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_RESOURCE_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_RESOURCE_SITE}
        ${X11PROTO_RESOURCE_FULLNAME}
        ${X11PROTO_RESOURCE_BASE}
        ${X11PROTO_RESOURCE_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}")
    add_dependencies(${X11PROTO_RESOURCE_BASE} libX11-1.6.4)
else()
    add_custom_target(${X11PROTO_RESOURCE_BASE})
endif()
