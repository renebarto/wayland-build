project(scrnsaverproto)

set(X11PROTO_SCRNSAVER_VERSION 1.2.2)
set(X11PROTO_SCRNSAVER_REQUIRED_VERSION ${X11PROTO_SCRNSAVER_VERSION})

find_package(X11ProtoScrnSaver QUIET)

set(X11PROTO_SCRNSAVER_NAME ${PROJECT_NAME})
set(X11PROTO_SCRNSAVER_BASE ${X11PROTO_SCRNSAVER_NAME}-${X11PROTO_SCRNSAVER_VERSION})

if (NOT X11PROTO_SCRNSAVER_FOUND)
    set(X11PROTO_SCRNSAVER_EXTENSION tar.gz)
    set(X11PROTO_SCRNSAVER_FULLNAME ${X11PROTO_SCRNSAVER_BASE}.${X11PROTO_SCRNSAVER_EXTENSION})
    set(X11PROTO_SCRNSAVER_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_SCRNSAVER_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_SCRNSAVER_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_SCRNSAVER_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_SCRNSAVER_SITE}
        ${X11PROTO_SCRNSAVER_FULLNAME}
        ${X11PROTO_SCRNSAVER_BASE}
        ${X11PROTO_SCRNSAVER_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}")
    add_dependencies(${X11PROTO_SCRNSAVER_BASE} libX11-1.6.4)
else()
    add_custom_target(${X11PROTO_SCRNSAVER_BASE})
endif()
