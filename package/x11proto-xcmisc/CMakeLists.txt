project(xcmiscproto)

set(X11PROTO_XCMISC_VERSION 1.2.2)
set(X11PROTO_XCMISC_REQUIRED_VERSION ${X11PROTO_XCMISC_VERSION})

find_package(X11ProtoXcmisc QUIET)

set(X11PROTO_XCMISC_NAME ${PROJECT_NAME})
set(X11PROTO_XCMISC_BASE ${X11PROTO_XCMISC_NAME}-${X11PROTO_XCMISC_VERSION})

if (NOT X11PROTO_XCMISC_FOUND)
    set(X11PROTO_XCMISC_EXTENSION tar.gz)
    set(X11PROTO_XCMISC_FULLNAME ${X11PROTO_XCMISC_BASE}.${X11PROTO_XCMISC_EXTENSION})
    set(X11PROTO_XCMISC_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_XCMISC_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_XCMISC_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_XCMISC_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_XCMISC_SITE}
        ${X11PROTO_XCMISC_FULLNAME}
        ${X11PROTO_XCMISC_BASE}
        ${X11PROTO_XCMISC_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}"
        "make"
        "make check"
        "make install")
else()
    add_custom_target(${X11PROTO_XCMISC_BASE})
endif()
