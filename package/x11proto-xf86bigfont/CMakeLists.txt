project(xf86bigfontproto)

set(X11PROTO_XF86BIGFONT_VERSION 1.2.0)
set(X11PROTO_XF86BIGFONT_REQUIRED_VERSION ${X11PROTO_XF86BIGFONT_VERSION})

find_package(X11ProtoXf86BigFont QUIET)

set(X11PROTO_XF86BIGFONT_NAME ${PROJECT_NAME})
set(X11PROTO_XF86BIGFONT_BASE ${X11PROTO_XF86BIGFONT_NAME}-${X11PROTO_XF86BIGFONT_VERSION})

if (NOT X11PROTO_XF86BIGFONT_FOUND)
    set(X11PROTO_XF86BIGFONT_EXTENSION tar.gz)
    set(X11PROTO_XF86BIGFONT_FULLNAME ${X11PROTO_XF86BIGFONT_BASE}.${X11PROTO_XF86BIGFONT_EXTENSION})
    set(X11PROTO_XF86BIGFONT_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_XF86BIGFONT_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_XF86BIGFONT_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_XF86BIGFONT_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_XF86BIGFONT_SITE}
        ${X11PROTO_XF86BIGFONT_FULLNAME}
        ${X11PROTO_XF86BIGFONT_BASE}
        ${X11PROTO_XF86BIGFONT_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}"
        "make"
        "make check"
        "make install")
else()
    add_custom_target(${X11PROTO_XF86BIGFONT_BASE})
endif()
