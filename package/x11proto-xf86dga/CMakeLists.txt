project(xf86dgaproto)

set(X11PROTO_XF86DGA_VERSION 2.1)
set(X11PROTO_XF86DGA_REQUIRED_VERSION ${X11PROTO_XF86DGA_VERSION})

find_package(X11ProtoXf86DGA QUIET)

set(X11PROTO_XF86DGA_NAME ${PROJECT_NAME})
set(X11PROTO_XF86DGA_BASE ${X11PROTO_XF86DGA_NAME}-${X11PROTO_XF86DGA_VERSION})

if (NOT X11PROTO_XF86DGA_FOUND)
    set(X11PROTO_XF86DGA_EXTENSION tar.gz)
    set(X11PROTO_XF86DGA_FULLNAME ${X11PROTO_XF86DGA_BASE}.${X11PROTO_XF86DGA_EXTENSION})
    set(X11PROTO_XF86DGA_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_XF86DGA_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_XF86DGA_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_XF86DGA_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_XF86DGA_SITE}
        ${X11PROTO_XF86DGA_FULLNAME}
        ${X11PROTO_XF86DGA_BASE}
        ${X11PROTO_XF86DGA_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}")
else()
    add_custom_target(${X11PROTO_XF86DGA_BASE})
endif()
