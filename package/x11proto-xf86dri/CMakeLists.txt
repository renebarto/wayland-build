project(xf86driproto)

set(X11PROTO_XF86DRI_VERSION 2.1.1)
set(X11PROTO_XF86DRI_REQUIRED_VERSION ${X11PROTO_XF86DRI_VERSION})

find_package(X11ProtoXf86DRI QUIET)

set(X11PROTO_XF86DRI_NAME ${PROJECT_NAME})
set(X11PROTO_XF86DRI_BASE ${X11PROTO_XF86DRI_NAME}-${X11PROTO_XF86DRI_VERSION})

if (NOT X11PROTO_XF86DRI_FOUND)
    set(X11PROTO_XF86DRI_EXTENSION tar.gz)
    set(X11PROTO_XF86DRI_FULLNAME ${X11PROTO_XF86DRI_BASE}.${X11PROTO_XF86DRI_EXTENSION})
    set(X11PROTO_XF86DRI_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_XF86DRI_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_XF86DRI_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_XF86DRI_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_XF86DRI_SITE}
        ${X11PROTO_XF86DRI_FULLNAME}
        ${X11PROTO_XF86DRI_BASE}
        ${X11PROTO_XF86DRI_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}"
        "make"
        "make check"
        "make install")
else()
    add_custom_target(${X11PROTO_XF86DRI_BASE})
endif()
