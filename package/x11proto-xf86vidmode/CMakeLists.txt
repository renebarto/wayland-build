project(xf86vidmodeproto)

set(X11PROTO_XF86VIDMODE_VERSION 2.3.1)
set(X11PROTO_XF86VIDMODE_REQUIRED_VERSION ${X11PROTO_XF86VIDMODE_VERSION})

find_package(X11ProtoXf86VidMode QUIET)

set(X11PROTO_XF86VIDMODE_NAME ${PROJECT_NAME})
set(X11PROTO_XF86VIDMODE_BASE ${X11PROTO_XF86VIDMODE_NAME}-${X11PROTO_XF86VIDMODE_VERSION})

if (NOT X11PROTO_XF86VIDMODE_FOUND)
    set(X11PROTO_XF86VIDMODE_EXTENSION tar.gz)
    set(X11PROTO_XF86VIDMODE_FULLNAME ${X11PROTO_XF86VIDMODE_BASE}.${X11PROTO_XF86VIDMODE_EXTENSION})
    set(X11PROTO_XF86VIDMODE_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_XF86VIDMODE_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_XF86VIDMODE_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_XF86VIDMODE_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_XF86VIDMODE_SITE}
        ${X11PROTO_XF86VIDMODE_FULLNAME}
        ${X11PROTO_XF86VIDMODE_BASE}
        ${X11PROTO_XF86VIDMODE_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}")
    add_dependencies(${X11PROTO_XF86VIDMODE_BASE} libX11-1.6.4)
else()
    add_custom_target(${X11PROTO_XF86VIDMODE_BASE})
endif()
