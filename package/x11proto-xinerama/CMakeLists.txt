project(xineramaproto)

set(X11PROTO_XINERAMA_VERSION 1.2.1)
set(X11PROTO_XINERAMA_REQUIRED_VERSION ${X11PROTO_XINERAMA_VERSION})

find_package(X11ProtoXinerama QUIET)

set(X11PROTO_XINERAMA_NAME ${PROJECT_NAME})
set(X11PROTO_XINERAMA_BASE ${X11PROTO_XINERAMA_NAME}-${X11PROTO_XINERAMA_VERSION})

if (NOT X11PROTO_XINERAMA_FOUND)
    set(X11PROTO_XINERAMA_EXTENSION tar.gz)
    set(X11PROTO_XINERAMA_FULLNAME ${X11PROTO_XINERAMA_BASE}.${X11PROTO_XINERAMA_EXTENSION})
    set(X11PROTO_XINERAMA_SITE ftp://ftp.x.org/pub/individual/proto)
    set(X11PROTO_XINERAMA_STAMP_DIR ${PACKAGE_STAMP_DIR}/${X11PROTO_XINERAMA_BASE})
    setup_package_automake(
        ${PACKAGE_DOWNLOAD_DIR}
        ${PACKAGE_BUILD_DIR}
        ${PACKAGE_STAGING_DIR}
        ${X11PROTO_XINERAMA_STAMP_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${X11PROTO_XINERAMA_SITE}
        ${X11PROTO_XINERAMA_FULLNAME}
        ${X11PROTO_XINERAMA_BASE}
        ${X11PROTO_XINERAMA_BASE}
        ""
        ""
        "./configure --prefix=${CMAKE_INSTALL_PREFIX}")
    add_dependencies(${X11PROTO_XINERAMA_BASE} libX11-1.6.4)
else()
    add_custom_target(${X11PROTO_XINERAMA_BASE})
endif()
